## API Service Layer (Single Source of Truth)

All HTTP calls go through `services/api.js`:

```javascript
// ✓ Always use the API service
const response = await usersApi.getProfile();

// ✗ Never fetch directly
const response = await fetch('/api/users/profile');
```

### Token Management (In-Memory)

Tokens are stored **in memory** (never localStorage):

```javascript
import { setToken, getToken, clearToken } from '../services/api'

// On login
setToken(token)

// On logout
clearToken()
```

Session persistence: `sessionStorage` stores `tt_token` and `tt_user` (cleared when tab closes).

### Adding New API Calls

In `services/api.js`:

```javascript
export const myApi = {
  // GET /api/my/resource
  getAll: () => request('/my/resource'),
  
  // GET /api/my/resource/:id
  getOne: (id) => request(`/my/resource/${id}`),
  
  // POST /api/my/resource
  create: (data) => request('/my/resource', { 
    method: 'POST', 
    body: JSON.stringify(data) 
  }),
  
  // PUT /api/my/resource/:id
  update: (id, data) => request(`/my/resource/${id}`, { 
    method: 'PUT', 
    body: JSON.stringify(data) 
  }),
  
  // DELETE /api/my/resource/:id
  delete: (id) => request(`/my/resource/${id}`, { method: 'DELETE' }),
};
```

Then use in components:

```javascript
async function handleFetch() {
  try {
    const data = await myApi.getAll();
    setItems(data);
  } catch (err) {
    toast(err.message, 'error');
  }
}
```

## State Management with Context API

### Auth Context Pattern

```javascript
// context/AuthContext.jsx
import { createContext, useContext, useState, useEffect } from 'react';
import { authApi, setToken, clearToken } from '../services/api';

const AuthContext = createContext(null);

export function AuthProvider({ children }) {
  const [user, setUser] = useState(null);
  const [token, setAuthToken] = useState(null);
  const [loading, setLoading] = useState(true);

  // Restore session on mount
  useEffect(() => {
    const stored = sessionStorage.getItem('tt_token');
    if (stored) {
      setToken(stored);
      setAuthToken(stored);
    }
    setLoading(false);
  }, []);

  const login = async (email, password) => {
    const { token, user } = await authApi.login({ email, password });
    setToken(token);
    setAuthToken(token);
    setUser(user);
    sessionStorage.setItem('tt_token', token);
    sessionStorage.setItem('tt_user', JSON.stringify(user));
  };

  const logout = () => {
    clearToken();
    setAuthToken(null);
    setUser(null);
    sessionStorage.removeItem('tt_token');
    sessionStorage.removeItem('tt_user');
  };

  return (
    <AuthContext.Provider value={{ user, token, login, logout, loading }}>
      {children}
    </AuthContext.Provider>
  );
}

export function useAuth() {
  const ctx = useContext(AuthContext);
  if (!ctx) throw new Error('useAuth must be used within AuthProvider');
  return ctx;
}
```

### Other Contexts (Cart, Wishlist, etc.)

Keep simple and focused:

```javascript
// context/CartContext.jsx
export function CartProvider({ children }) {
  const [cart, setCart] = useState([]);

  const addItem = (item) => setCart([...cart, item]);
  const removeItem = (id) => setCart(cart.filter(item => item.id !== id));
  const clear = () => setCart([]);

  return (
    <CartContext.Provider value={{ cart, addItem, removeItem, clear }}>
      {children}
    </CartContext.Provider>
  );
}

export function useCart() {
  const ctx = useContext(CartContext);
  if (!ctx) throw new Error('useCart must be used within CartProvider');
  return ctx;
}
```

### Using Contexts in Components

```javascript
function MyComponent() {
  const { user, logout } = useAuth();
  const { cart, removeItem } = useCart();

  return (
    <div>
      <p>Hello, {user.name}</p>
      <p>Cart items: {cart.length}</p>
      <button onClick={logout}>Logout</button>
    </div>
  );
}
```

## Component Structure

### Page Component Template

```javascript
// pages/MyPage.jsx
import { useState, useEffect } from 'react';
import { myApi } from '../services/api';
import { useToast } from '../hooks/useToast';
import MyComponent from '../components/MyComponent';

export default function MyPage() {
  const [items, setItems] = useState([]);
  const [loading, setLoading] = useState(false);
  const { toast } = useToast();

  useEffect(() => {
    async function fetchData() {
      setLoading(true);
      try {
        const data = await myApi.getAll();
        setItems(data);
      } catch (err) {
        toast(err.message, 'error');
      } finally {
        setLoading(false);
      }
    }
    fetchData();
  }, []);

  if (loading) return <div>Loading...</div>;

  return (
    <div>
      {items.map(item => (
        <MyComponent key={item.id} item={item} />
      ))}
    </div>
  );
}
```

### Reusable Component Template

```javascript
// components/MyComponent.jsx
import { useState } from 'react';
import { useToast } from '../hooks/useToast';

export default function MyComponent({ item, onDelete }) {
  const [editing, setEditing] = useState(false);
  const { toast } = useToast();

  const handleDelete = async () => {
    try {
      await myApi.delete(item.id);
      onDelete?.(item.id);
      toast('Deleted successfully', 'success');
    } catch (err) {
      toast(err.message, 'error');
    }
  };

  return (
    <div className="my-component">
      <h3>{item.name}</h3>
      <button onClick={handleDelete}>Delete</button>
    </div>
  );
}
```

## Routing (Custom View-Based)

Views are controlled by state, not a routing library:

```javascript
// App.jsx
const [view, setView] = useState('landing');

return (
  <>
    {view === 'landing' && <Landing onGoStore={() => setView('store')} />}
    {view === 'store' && <Store onGoCartCheckout={() => setView('checkout')} />}
    {view === 'checkout' && <Checkout />}
    {view === 'user' && <DashboardUser />}
    {view === 'org' && <DashboardOrg />}
    {view === 'admin' && <DashboardAdmin />}
  </>
);
```

## Toast Notifications

Use the `useToast` hook:

```javascript
import { useToast } from '../hooks/useToast';

function MyComponent() {
  const { toast } = useToast();

  const handleAction = async () => {
    try {
      await someApi.call();
      toast('Success!', 'success');
    } catch (err) {
      toast(`Error: ${err.message}`, 'error');
    }
  };

  return <button onClick={handleAction}>Do Something</button>;
}
```

Toast types: `success`, `error`, `info` (default style).

## Common Patterns

### Conditional Rendering Based on Auth
```javascript
function MyPage() {
  const { user } = useAuth();
  
  if (!user) return <p>Please log in</p>;
  if (user.role !== 'admin') return <p>Admin only</p>;
  
  return <AdminPanel />;
}
```

### Loading States
```javascript
function MyComponent() {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(false);

  useEffect(() => {
    (async () => {
      setLoading(true);
      try {
        const result = await api.fetch();
        setData(result);
      } finally {
        setLoading(false);
      }
    })();
  }, []);

  return loading ? <Spinner /> : <Content data={data} />;
}
```

### Form Handling
```javascript
function LoginForm() {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const { login } = useAuth();
  const { toast } = useToast();

  const handleSubmit = async (e) => {
    e.preventDefault();
    try {
      await login(email, password);
      toast('Logged in!', 'success');
    } catch (err) {
      toast(err.message, 'error');
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <input value={email} onChange={(e) => setEmail(e.target.value)} />
      <input type="password" value={password} onChange={(e) => setPassword(e.target.value)} />
      <button type="submit">Login</button>
    </form>
  );
}
```
