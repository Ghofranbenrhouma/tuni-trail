## Database Patterns

### Connection Pool
```javascript
const pool = require('../config/db'); // mysql2/promise
const [rows] = await pool.query('SELECT * FROM users WHERE id = ?', [userId]);
```

**Remember**: 
- Always destructure `[rows]` (metadata is second element)
- Use parameterized queries for safety
- Connection limit: 10 (configured in db.js)

### Transaction Example
```javascript
const conn = await pool.getConnection();
try {
  await conn.beginTransaction();
  await conn.query('UPDATE foo SET x = ?', [val1]);
  await conn.query('INSERT INTO bar VALUES (?)', [val2]);
  await conn.commit();
} catch (err) {
  await conn.rollback();
  throw err;
} finally {
  conn.release();
}
```

### Async/Await Pattern
**ALWAYS use async/await. NEVER chain `.then()`:**

```javascript
// ✓ Good
exports.getUser = async (req, res, next) => {
  try {
    const [rows] = await pool.query('SELECT * FROM users WHERE id = ?', [req.params.id]);
    res.json(rows[0]);
  } catch (err) {
    next(err);
  }
};

// ✗ Bad
exports.getUser = (req, res, next) => {
  pool.query(...).then(([rows]) => res.json(rows[0])).catch(next);
};
```

## Authentication & Authorization

### JWT Verification
The `auth` middleware extracts and verifies tokens:

```javascript
router.get('/profile', auth, controller.getProfile);
// req.user = { id, email, role }
```

### Role-Based Access Control
```javascript
router.delete('/admin-panel', auth, roles('admin'), controller.deletePanel);
// Only admins can access
```

### Demo Login
Special endpoint allows testing without credentials:
```javascript
POST /api/auth/login { "role": "org" }  // Logs in as org@demo.com
POST /api/auth/login { "role": "user" } // Logs in as user@demo.com
```

## Controller Structure

### Template
```javascript
const pool = require('../config/db');

// GET /api/resource
exports.getResource = async (req, res, next) => {
  try {
    const [rows] = await pool.query('SELECT * FROM resource');
    res.json(rows);
  } catch (err) {
    next(err);
  }
};

// POST /api/resource
exports.createResource = async (req, res, next) => {
  try {
    const { name } = req.body;
    if (!name) return res.status(400).json({ error: 'Name required' });

    const [result] = await pool.query('INSERT INTO resource (name) VALUES (?)', [name]);
    res.status(201).json({ success: true, id: result.insertId });
  } catch (err) {
    next(err);
  }
};

// DELETE /api/resource/:id
exports.deleteResource = async (req, res, next) => {
  try {
    const { id } = req.params;
    await pool.query('DELETE FROM resource WHERE id = ?', [id]);
    res.json({ success: true });
  } catch (err) {
    next(err);
  }
};
```

### Response Format
```javascript
// Success
res.json({ success: true, data: {...} })
res.status(201).json({ success: true, id: result.insertId })

// Error
res.status(400).json({ error: 'Invalid input' })
res.status(401).json({ error: 'Unauthorized' })
res.status(403).json({ error: 'Forbidden' })
res.status(404).json({ error: 'Not found' })

// Global error handler catches and formats
next(err); // Delegate to errorHandler middleware
```

## Routing Structure

1. **Create controller** (`controllers/fooController.js`):
   - Export async functions: `exports.getFoo`, `exports.postFoo`, etc.
   - All wrap in try/catch

2. **Create router** (`routes/foo.js`):
   ```javascript
   const express = require('express');
   const { auth, roles } = require('../middleware/auth');
   const { getFoo, postFoo } = require('../controllers/fooController');
   
   const router = express.Router();
   router.get('/', getFoo);
   router.post('/', auth, postFoo);
   router.delete('/:id', auth, roles('admin'), deleteFoo);
   module.exports = router;
   ```

3. **Register in server.js**:
   ```javascript
   app.use('/api/foo', require('./routes/foo'));
   ```

## Debugging Checklist

- ✓ Import pool: `const pool = require('../config/db')`
- ✓ Always destructure: `const [rows] = await pool.query(...)`
- ✓ Use parameterized queries: `pool.query(sql, [param1, param2])`
- ✓ Wrap in try/catch: `catch (err) { next(err); }`
- ✓ Set Authorization header in requests: `Authorization: Bearer <token>`
- ✓ Verify JWT_SECRET matches across auth & verification
- ✓ Check CORS config in server.js for frontend origin

## Next: OpenAPI/Swagger Documentation

To add API spec & interactive UI docs:

1. Install tools:
   ```bash
   npm install swagger-ui-express swagger-jsdoc
   ```

2. Create `swagger.js`:
   ```javascript
   const swaggerJsdoc = require('swagger-jsdoc');
   const swaggerUi = require('swagger-ui-express');

   const options = {
     definition: {
       openapi: '3.0.0',
       info: { title: 'TuniTrail API', version: '1.0.0' },
       servers: [{ url: 'http://localhost:5000/api' }],
     },
     apis: ['./routes/*.js'],
   };

   const specs = swaggerJsdoc(options);
   module.exports = { swaggerUi, specs };
   ```

3. In `server.js`:
   ```javascript
   const { swaggerUi, specs } = require('./swagger');
   app.use('/docs', swaggerUi.serve, swaggerUi.setup(specs));
   ```

4. Add JSDoc to routes:
   ```javascript
   /**
    * @openapi
    * /api/events:
    *   get:
    *     summary: List all events
    *     responses:
    *       200:
    *         description: Array of events
    */
   router.get('/', controller.getEvents);
   ```

Accessible at: `http://localhost:5000/docs`
